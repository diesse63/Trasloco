CREATE TABLE public.mobili (
  id bigint GENERATED ALWAYS AS IDENTITY NOT NULL,
  idstanza bigint NOT NULL,
  nome text NOT NULL,
  note text,
  foto text,
  CONSTRAINT mobili_pkey PRIMARY KEY (id),
  CONSTRAINT mobili_idstanza_fkey FOREIGN KEY (idstanza) REFERENCES public.stanze(id)
);
CREATE TABLE public.oggetti (
  id bigint GENERATED ALWAYS AS IDENTITY NOT NULL,
  nome text,
  idscatola bigint NOT NULL,
  idstanza bigint NOT NULL,
  idmobile bigint,
  pathfoto text NOT NULL,
  thumbnail text,
  note text,
  CONSTRAINT oggetti_pkey PRIMARY KEY (id),
  CONSTRAINT oggetti_idscatola_fkey FOREIGN KEY (idscatola) REFERENCES public.scatole(id),
  CONSTRAINT oggetti_idstanza_fkey FOREIGN KEY (idstanza) REFERENCES public.stanze(id),
  CONSTRAINT oggetti_idmobile_fkey FOREIGN KEY (idmobile) REFERENCES public.mobili(id)
);
CREATE TABLE public.scatole (
  id bigint GENERATED ALWAYS AS IDENTITY NOT NULL,
  nome text UNIQUE,
  pathfoto text,
  note text,
  data timestamp without time zone,
  stato boolean NOT NULL DEFAULT false,
  CONSTRAINT scatole_pkey PRIMARY KEY (id)
);
CREATE TABLE public.stanze (
  id bigint GENERATED ALWAYS AS IDENTITY NOT NULL,
  nome text NOT NULL UNIQUE,
  note text,
  CONSTRAINT stanze_pkey PRIMARY KEY (id)
);
